        function getRandomNode(nodeArray) {
            return nodeArray[Math.floor(Math.random() * nodeArray.length)];
        }

        function distToSegment(p, v, w) {
            let l2 = (v.x - w.x)**2 + (v.y - w.y)**2;
            if (l2 === 0) return Math.hypot(p.x - v.x, p.y - v.y);
            let t = ((p.x - v.x) * (w.x - v.x) + (p.y - v.y) * (w.y - v.y)) / l2;
            t = Math.max(0, Math.min(1, t));
            return Math.hypot(p.x - (v.x + t * (w.x - v.x)), p.y - (v.y + t * (w.y - v.y)));
        }

        function isPath(x, y) {
            let p = {x, y};
            for (let seg of pathSegments) {
                if (distToSegment(p, seg.p1, seg.p2) < PATH_WIDTH) {
                    return true;
                }
            }
            return false;
        }

        function generateWave(waveNum) {
            let queue = [];
            let numCreeps = 10 + waveNum * 2; 
            let isBossWave = (waveNum % 10 === 0);

            if (isBossWave) numCreeps += 2; 

            for (let i = 0; i < numCreeps; i++) {
                let startNode = getRandomNode(Nodes.Starts);
                
                let isHeavy = (waveNum >= 4 && i % 4 === 0); 
                let isFast = (waveNum >= 7 && i % 5 === 0 && !isHeavy);
                let isSiege = (waveNum >= 8 && i % 10 === 0);

                if (isBossWave && i === numCreeps - 1) {
                    queue.push({ type: 'boss', delay: 150, startNode: startNode });
                } else if (isSiege) {
                    queue.push({ type: 'siege', delay: 80, startNode: startNode }); 
                } else if (isHeavy) {
                    queue.push({ type: 'heavy', delay: 50, startNode: startNode }); 
                } else if (isFast) {
                    queue.push({ type: 'fast', delay: 20, startNode: startNode }); 
                } else {
                    queue.push({ type: 'normal', delay: 35, startNode: startNode }); 
                }
            }
            if (queue.length > 0) queue[0].delay = 0;
            return queue;
        }

        class FloatingText {
            constructor(x, y, text, color) {
                this.x = x; this.y = y; this.text = text; this.color = color;
                this.life = 40; this.maxLife = 40;
            }
            update() { this.y -= 1.5; this.life--; }
            draw(ctx) {
                ctx.fillStyle = this.color;
                ctx.globalAlpha = this.life / this.maxLife;
                ctx.font = "bold 48px Arial";
                ctx.fillText(this.text, this.x, this.y);
                ctx.globalAlpha = 1.0;
            }
        }

        class HealBeamEffect {
            constructor(x1, y1, target, amount) {
                this.x1 = x1; this.y1 = y1; this.target = target;
                this.life = 15; this.maxLife = 15;
            }
            update() { this.life--; }
            draw(ctx) {
                ctx.strokeStyle = "#e040fb"; ctx.lineWidth = 12;
                ctx.globalAlpha = this.life / this.maxLife;
                ctx.beginPath(); ctx.moveTo(this.x1, this.y1); ctx.lineTo(this.target.x, this.target.y); ctx.stroke();
                ctx.globalAlpha = 1.0;
            }
        }

        class AxeAttackEffect {
            constructor(x, y, radius) {
                this.x = x; this.y = y; this.radius = radius;
                this.life = 15; this.maxLife = 15;
            }
            update() { this.life--; }
            draw(ctx) {
                ctx.strokeStyle = "#ff5252"; ctx.lineWidth = 10;
                ctx.globalAlpha = this.life / this.maxLife;
                ctx.beginPath(); ctx.arc(this.x, this.y, this.radius * (1 - this.life/this.maxLife), 0, Math.PI * 2); ctx.stroke();
                ctx.globalAlpha = 1.0;
            }
        }

        class Creep {
            constructor(waveParams) {
                this.path = [
                    waveParams.startNode, Nodes.I1, 
                    getRandomNode(Nodes.Mid), Nodes.I2, 
                    getRandomNode(Nodes.End), Nodes.I3, 
                    Nodes.Throne
                ];
                this.currentWaypoint = 0;
                this.waveType = waveParams.type;
                
                this.offsetX = (Math.random() - 0.5) * (PATH_WIDTH * 0.8);
                this.offsetY = (Math.random() - 0.5) * (PATH_WIDTH * 0.8);
                
                this.x = this.path[0].x + this.offsetX;
                this.y = this.path[0].y + this.offsetY;
                
                this.isBoss = false;
                this.reachedEnd = false;

                if (waveParams.type === 'boss') {
                    this.hp = 2000 + (currentWave * 300); 
                    this.speed = 0.5 * SCALE; this.reward = 150; this.radius = 35 * SCALE; this.color = "#b71c1c"; this.damage = 150; 
                    this.isBoss = true;
                } else if (waveParams.type === 'heavy') {
                    this.hp = 300 + (currentWave * 70);
                    this.speed = 0.8 * SCALE; this.reward = 15; this.radius = 20 * SCALE; this.color = "#8e24aa"; this.damage = 25;
                } else if (waveParams.type === 'fast') {
                    this.hp = 50 + (currentWave * 12);
                    this.speed = (2.5 + (currentWave * 0.05)) * SCALE; this.reward = 6; this.radius = 8 * SCALE; this.color = "#00bcd4"; this.damage = 5;
                } else if (waveParams.type === 'siege') {
                    this.hp = 500 + (currentWave * 80);
                    this.speed = 0.6 * SCALE; this.reward = 25; this.radius = 25 * SCALE; this.color = "#ffeb3b"; this.damage = 50;
                } else {
                    this.hp = 80 + (currentWave * 15);
                    this.speed = (1.2 + (currentWave * 0.05)) * SCALE; this.reward = 8; this.radius = 12 * SCALE; this.color = "#ffb74d"; this.damage = 10;
                }
                
                this.maxHp = this.hp; this.baseSpeed = this.speed;
                this.slowTimer = 0;
                this.blockedBy = null;
                this.attackTimer = 0;
            }

            update() {
                if (this.slowTimer > 0) { this.slowTimer--; } else { this.speed = this.baseSpeed; }

                if (this.blockedBy) {
                    let distToAxe = Math.hypot(this.blockedBy.x - this.x, this.blockedBy.y - this.y);
                    if (this.blockedBy.hp <= 0 || distToAxe > (60 * SCALE)) {
                        this.blockedBy = null;
                    }
                }

                if (!this.blockedBy) {
                    for (let t of towers) {
                        if (t.typeInfo.type === 'melee' && t.hp > 0) {
                            let dist = Math.hypot(t.x - this.x, t.y - this.y);
                            if (dist < 45 * SCALE) {
                                this.blockedBy = t;
                                break;
                            }
                        }
                    }
                }

                if (this.blockedBy) {
                    if (this.attackTimer <= 0) {
                        this.blockedBy.hp -= this.damage; 
                        visualEffects.push(new FloatingText(this.blockedBy.x, this.blockedBy.y - 40, `-${this.damage}`, '#ff0000'));
                        this.attackTimer = 60; 
                    } else {
                        this.attackTimer--;
                    }
                    return; 
                }

                let target = this.path[this.currentWaypoint + 1];
                if (!target) {
                    this.hp = 0;
                    this.reachedEnd = true;
                    
                    throneHp -= this.damage;
                    if (throneHp < 0) throneHp = 0;
                    
                    document.getElementById('throne-hp-display').innerText = throneHp;
                    visualEffects.push(new FloatingText(this.x, this.y, `-${this.damage} Трон!`, '#ff0000'));
                    
                    if (throneHp <= 0) {
                        gameOver = true;
                        document.getElementById('game-over-modal').style.display = 'flex';
                    }
                    return;
                }
                
                let targetX = target.x + this.offsetX;
                let targetY = target.y + this.offsetY;
                
                let dx = targetX - this.x;
                let dy = targetY - this.y;
                let dist = Math.hypot(dx, dy);

                if (dist < this.speed) {
                    this.x = targetX;
                    this.y = targetY;
                    this.currentWaypoint++;
                } else {
                    this.x += (dx / dist) * this.speed;
                    this.y += (dy / dist) * this.speed;
                }
            }

            draw(ctx) {
                let sprite;
                if (this.waveType === 'boss') sprite = images.boss;
                else if (this.waveType === 'heavy' || this.waveType === 'siege') sprite = images.creep_fat;
                else if (this.waveType === 'fast') sprite = images.wolf;
                else sprite = images.creep_swordsman;

                if (sprite && sprite.complete && sprite.naturalWidth !== 0) {
                    let size = this.radius * 2.8;
                    
                    if (this.slowTimer > 0) {
                        ctx.fillStyle = "rgba(132, 255, 255, 0.4)";
                        ctx.beginPath(); ctx.arc(this.x, this.y, this.radius, 0, Math.PI * 2); ctx.fill();
                    }

                    ctx.drawImage(sprite, this.x - size/2, this.y - size/2, size, size);

                    if (this.isBoss) {
                        ctx.fillStyle = "white"; ctx.font = "bold 36px Arial"; ctx.textAlign = "center";
                        ctx.fillText("BOSS", this.x, this.y - size/2 - 10);
                    }
                } else {
                    ctx.fillStyle = (this.slowTimer > 0) ? "#84ffff" : this.color;
                    ctx.beginPath();
                    ctx.arc(this.x, this.y, this.radius, 0, Math.PI * 2);
                    ctx.fill();
                    ctx.strokeStyle = "#333";
                    ctx.lineWidth = 4;
                    ctx.stroke();

                    if (this.isBoss) {
                        ctx.fillStyle = "white"; ctx.font = "bold 36px Arial"; ctx.textAlign = "center";
                        ctx.fillText("BOSS", this.x, this.y + 12);
                    }
                }

                // Health bar
                const barWidth = 120;
                const barHeight = 16;
                ctx.fillStyle = "red"; ctx.fillRect(this.x - barWidth/2, this.y - this.radius - 30, barWidth, barHeight);
                ctx.fillStyle = "#4CAF50"; ctx.fillRect(this.x - barWidth/2, this.y - this.radius - 30, barWidth * (this.hp / this.maxHp), barHeight);
            }
        }

        class Tower {
            constructor(x, y, typeInfo) {
                this.x = x; this.y = y;
                this.rallyX = x; this.rallyY = y; 
                this.hp = typeInfo.maxHp || 0;
                this.maxHp = typeInfo.maxHp || 0;
                this.typeInfo = typeInfo;
                this.cooldownTimer = 0;
            }

            findTarget() {
                for (let c of creeps) {
                    if (Math.hypot(c.x - this.x, c.y - this.y) <= this.typeInfo.range) return c;
                }
                return null;
            }

            findMeleeTargets() {
                let targets = [];
                for (let c of creeps) {
                    if (Math.hypot(c.x - this.rallyX, c.y - this.rallyY) <= this.typeInfo.engagementRange) targets.push(c);
                }
                return targets;
            }

            findHealTarget() {
                let lowestHpAlly = null;
                let lowestHpPercent = 1.0;
                for (let t of towers) {
                    if (t.typeInfo.type === 'melee' && t.hp > 0 && t.hp < t.maxHp) {
                        if (Math.hypot(t.x - this.x, t.y - this.y) <= this.typeInfo.range) {
                            let hpPercent = t.hp / t.maxHp;
                            if (hpPercent < lowestHpPercent) {
                                lowestHpPercent = hpPercent;
                                lowestHpAlly = t;
                            }
                        }
                    }
                }
                return lowestHpAlly;
            }

            update() {
                if (this.typeInfo.type === 'melee') {
                    let targets = this.findMeleeTargets();
                    if (targets.length > 0) {
                        let target = targets[0]; 
                        let dx = target.x - this.x; let dy = target.y - this.y; let dist = Math.hypot(dx, dy);
                        if (dist > 60) { 
                            this.x += (dx / dist) * this.typeInfo.speed; 
                            this.y += (dy / dist) * this.typeInfo.speed; 
                        }
                        
                        if (this.cooldownTimer === 0) {
                            targets.forEach(t => {
                                if (Math.hypot(t.x - this.x, t.y - this.y) <= 80 * SCALE) t.hp -= this.typeInfo.damage;
                            });
                            visualEffects.push(new AxeAttackEffect(this.x, this.y, 80 * SCALE));
                            this.cooldownTimer = this.typeInfo.fireRate;
                        }
                    } else {
                        let dx = this.rallyX - this.x; let dy = this.rallyY - this.y; let dist = Math.hypot(dx, dy);
                        if (dist > this.typeInfo.speed) { 
                            this.x += (dx / dist) * this.typeInfo.speed; 
                            this.y += (dy / dist) * this.typeInfo.speed; 
                        } else { 
                            this.x = this.rallyX; this.y = this.rallyY; 
                        }
                    }
                    if (this.cooldownTimer > 0) this.cooldownTimer--;

                } else if (this.typeInfo.type === 'healer') {
                    if (waveActive) {
                        if (this.cooldownTimer > 0) this.cooldownTimer--;
                        if (this.cooldownTimer === 0) {
                            let target = this.findHealTarget();
                            if (target) { 
                                target.hp = Math.min(target.maxHp, target.hp + this.typeInfo.healAmount); 
                                visualEffects.push(new HealBeamEffect(this.x, this.y - 40, target, this.typeInfo.healAmount)); 
                                visualEffects.push(new FloatingText(target.x, target.y - 80, `+${this.typeInfo.healAmount}`, '#00ff00'));
                                this.cooldownTimer = this.typeInfo.fireRate; 
                            }
                        }
                    }
                } else {
                    if (this.cooldownTimer > 0) this.cooldownTimer--;
                    if (this.cooldownTimer === 0) {
                        let target = this.findTarget();
                        if (target) {
                            // Огонь из центра героя
                            projectiles.push(new Projectile(this.x, this.y - 40, target, this.typeInfo));
                            this.cooldownTimer = this.typeInfo.fireRate;
                        }
                    }
                }
            }

            draw(ctx) {
                if (this.typeInfo.type === 'melee') {
                    ctx.strokeStyle = "rgba(255, 255, 255, 0.1)";
                    ctx.lineWidth = 4;
                    ctx.beginPath(); ctx.arc(this.rallyX, this.rallyY, this.typeInfo.engagementRange, 0, Math.PI * 2); ctx.stroke();
                } 

                let sprite = images[this.typeInfo.sprite];
                let size = this.typeInfo.radius * 2.5;

                if (sprite && sprite.complete && sprite.naturalWidth !== 0) {
                    ctx.drawImage(sprite, this.x - size/2, this.y - size/2, size, size);
                } else {
                    ctx.fillStyle = this.typeInfo.color;
                    ctx.beginPath(); ctx.arc(this.x, this.y, this.typeInfo.radius, 0, Math.PI * 2); ctx.fill();
                    ctx.strokeStyle = "#fff"; ctx.lineWidth = 6; ctx.stroke();
                }

                if (this.typeInfo.type === 'melee' && this.hp > 0 && this.maxHp > 0) {
                    ctx.fillStyle = "red"; ctx.fillRect(this.x - 40, this.y - this.typeInfo.radius - 20, 80, 10);
                    ctx.fillStyle = "#4CAF50"; ctx.fillRect(this.x - 40, this.y - this.typeInfo.radius - 20, 80 * (this.hp / this.maxHp), 10);
                }
            }
        }

        class Projectile {
            constructor(x, y, target, sourceInfo) {
                this.x = x; this.y = y; this.target = target; this.sourceInfo = sourceInfo;
            }
            update() {
                if (this.target.hp <= 0) return false;
                let dx = this.target.x - this.x; let dy = this.target.y - this.y;
                let dist = Math.hypot(dx, dy);
                if (dist < this.sourceInfo.projSpeed) {
                    this.target.hp -= this.sourceInfo.damage;
                    if (this.sourceInfo.slowFactor) {
                        this.target.speed = this.target.baseSpeed * this.sourceInfo.slowFactor;
                        this.target.slowTimer = this.sourceInfo.slowDuration;
                    }
                    return false; 
                }
                this.x += (dx / dist) * this.sourceInfo.projSpeed;
                this.y += (dy / dist) * this.sourceInfo.projSpeed;
                return true;
            }
            draw(ctx) {
                ctx.fillStyle = this.sourceInfo.color;
                ctx.shadowBlur = 30; ctx.shadowColor = this.sourceInfo.color;
                ctx.beginPath(); ctx.arc(this.x, this.y, 15, 0, Math.PI * 2); ctx.fill();
                ctx.shadowBlur = 0;
            }
        }

        function getCanvasCoordinates(e) {
            let rect = canvas.getBoundingClientRect();
            let scaleX = canvas.width / rect.width;
            let scaleY = canvas.height / rect.height;
            let x = (e.clientX - rect.left) * scaleX;
            let y = (e.clientY - rect.top) * scaleY;
            
            let gridX = Math.floor(x / GRID_SIZE) * GRID_SIZE + GRID_SIZE / 2;
            let gridY = Math.floor(y / GRID_SIZE) * GRID_SIZE + GRID_SIZE / 2;
            return { gridX, gridY };
        }

        canvas.addEventListener('click', function(e) {
            if (isPaused || gameOver || !gameStarted) return;
            let coords = getCanvasCoordinates(e);

            let typeInfo = TOWER_TYPES[selectedTowerType];
            let isOccupied = towers.some(t => t.x === coords.gridX && t.y === coords.gridY);

            if (gold >= typeInfo.cost && !isOccupied) {
                let isOnPath = isPath(coords.gridX, coords.gridY);
                if ((typeInfo.placeOnPath && isOnPath) || (!typeInfo.placeOnPath && !isOnPath)) {
                    gold -= typeInfo.cost;
                    document.getElementById('gold-display').innerText = gold;
                    towers.push(new Tower(coords.gridX, coords.gridY, typeInfo));
                }
            }
        });

        canvas.addEventListener('mousemove', function(e) {
            if (isPaused || gameOver || !gameStarted) return;
            let coords = getCanvasCoordinates(e);
            window.hoverX = coords.gridX;
            window.hoverY = coords.gridY;
        });

        function updateGame() {
            if (gameOver || !gameStarted) return;

            if (!waveActive) {
                waveTimer--;
                document.getElementById('wave-timer-display').innerText = `До волны: ${Math.ceil(waveTimer/60)} сек`;
                if (waveTimer <= 0) {
                    currentWave++;
                    waveActive = true;
                    spawnQueue = generateWave(currentWave);
                    document.getElementById('wave-display').innerText = currentWave;
                    document.getElementById('wave-timer-display').innerText = "Враги идут!";
                }
            } else {
                if (spawnQueue.length > 0) {
                    spawnTimer++;
                    if (spawnTimer >= spawnQueue[0].delay) {
                        creeps.push(new Creep(spawnQueue.shift()));
                        spawnTimer = 0;
                    }
                } else if (creeps.length === 0) {
                    waveActive = false;
                    waveTimer = 300; 
                }
            }

            towers.forEach(t => t.update());
            towers = towers.filter(t => t.typeInfo.type !== 'melee' || t.hp > 0); 
            projectiles = projectiles.filter(p => p.update());

            for (let i = creeps.length - 1; i >= 0; i--) {
                let c = creeps[i];
                c.update();
                if (c.hp <= 0 && !gameOver) {
                    if (!c.reachedEnd) {
                        gold += c.reward;
                        document.getElementById('gold-display').innerText = gold;
                        visualEffects.push(new FloatingText(c.x, c.y, `+${c.reward}g`, '#ffeb3b'));
                    }
                    creeps.splice(i, 1);
                }
            }

            visualEffects.forEach(v => v.update());
            visualEffects = visualEffects.filter(v => v.life > 0);
        }

        function drawPaths() {
            ctx.strokeStyle = '#222833';
            ctx.lineWidth = PATH_WIDTH * 2;
            ctx.lineJoin = 'round';
            ctx.lineCap = 'round';
            
            ctx.beginPath();
            pathSegments.forEach(seg => {
                ctx.moveTo(seg.p1.x, seg.p1.y);
                ctx.lineTo(seg.p2.x, seg.p2.y);
            });
            ctx.stroke();
        }

        function drawGame() {
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            ctx.fillStyle = '#11141a';
            ctx.fillRect(0, 0, canvas.width, canvas.height);

            if (images.map && images.map.complete && images.map.naturalWidth !== 0) {
                ctx.drawImage(images.map, 0, 0, canvas.width, canvas.height);
            } else {
                drawPaths();
            }

            Nodes.TowerSpots.forEach(spot => {
                ctx.fillStyle = 'rgba(0, 255, 204, 0.05)';
                ctx.strokeStyle = 'rgba(0, 255, 204, 0.3)';
                ctx.lineWidth = 6;
                ctx.fillRect(spot.x - 120, spot.y - 120, 240, 240);
                ctx.strokeRect(spot.x - 120, spot.y - 120, 240, 240);
            });

            // Трон (Radiant Base)
            if (images.throne && images.throne.complete && images.throne.naturalWidth !== 0) {
                let size = 450;
                ctx.drawImage(images.throne, Nodes.Throne.x - size/2, Nodes.Throne.y - size/2, size, size);
            } else {
                ctx.fillStyle = '#ffaa00';
                ctx.shadowBlur = 60; ctx.shadowColor = "red";
                ctx.beginPath(); ctx.arc(Nodes.Throne.x, Nodes.Throne.y, 160, 0, Math.PI*2); ctx.fill();
                ctx.strokeStyle = "red"; ctx.lineWidth = 10; ctx.stroke();
                ctx.shadowBlur = 0; 
            }

            if (window.hoverX && window.hoverY && !isPaused && !gameOver && gameStarted) {
                ctx.strokeStyle = "rgba(255, 255, 255, 0.02)";
                ctx.lineWidth = 2;
                ctx.beginPath();
                for (let x = 0; x <= canvas.width; x += GRID_SIZE) { ctx.moveTo(x, 0); ctx.lineTo(x, canvas.height); }
                for (let y = 0; y <= canvas.height; y += GRID_SIZE) { ctx.moveTo(0, y); ctx.lineTo(canvas.width, y); }
                ctx.stroke();
            }

            towers.forEach(t => t.draw(ctx));
            creeps.forEach(c => c.draw(ctx));
            projectiles.forEach(p => p.draw(ctx));
            visualEffects.forEach(v => v.draw(ctx));

            if (window.hoverX && window.hoverY && !isPaused && !gameOver && gameStarted) {
                let typeInfo = TOWER_TYPES[selectedTowerType];
                let isOnPath = isPath(window.hoverX, window.hoverY);
                let isOccupied = towers.some(t => t.x === window.hoverX && t.y === window.hoverY);
                let canPlace = ((typeInfo.placeOnPath && isOnPath) || (!typeInfo.placeOnPath && !isOnPath)) && !isOccupied;
                
                ctx.fillStyle = canPlace ? "rgba(0, 255, 0, 0.3)" : "rgba(255, 0, 0, 0.3)";
                ctx.fillRect(window.hoverX - GRID_SIZE/2, window.hoverY - GRID_SIZE/2, GRID_SIZE, GRID_SIZE);
                
                ctx.strokeStyle = canPlace ? "rgba(255, 255, 255, 0.5)" : "rgba(255, 0, 0, 0.5)";
                let range = (typeInfo.type === 'melee') ? typeInfo.engagementRange : typeInfo.range;
                ctx.beginPath(); ctx.arc(window.hoverX, window.hoverY, range, 0, Math.PI * 2); ctx.stroke();
            }
        }

        function gameLoop() {
            if (!isPaused) {
                updateGame();
            }
            drawGame();
            requestAnimationFrame(gameLoop);
        }

        gameLoop();
    </script>
